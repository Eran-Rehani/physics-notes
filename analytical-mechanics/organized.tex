%----------------------- I. Mechanics & Coordinates -----------------------
\textbf{I. Mechanics \& Coordinates}\\[2pt]
\textbf{Order of Ops:} 
\begin{enumerate}[noitemsep, topsep=0pt]
    \item \textbf{DOF:} Holonomic constraint 
    \[
    F(\vec{r}_1, \vec{r}_2, \ldots, \vec{r}_N, t)=0,
    \quad n=3N-k.
    \]
    \item Find \(n\) generalized coordinates.
    \item Express in terms of generalized coordinates.
    \item Compute Kinetic and Potential Energies.
    \item Write the Equations of Motion.
\end{enumerate}

\textbf{Gen. Kinetic Energy:}  
\[
T = \frac{1}{2} m (\dot{x}^2+\dot{y}^2+\dot{z}^2) \quad \text{(Cartesian)}
\]
\[
T = \frac{1}{2} m (\dot{r}^2+r^2\dot{\theta}^2+r^2\sin^2\theta\,\dot{\phi}^2) \quad \text{(Spherical)}
\]
\[
T = \frac{1}{2} m (\dot{r}^2+r^2\dot{\theta}^2+\dot{z}^2) \quad \text{(Cylindrical)}
\]

\textbf{(M \& K Matrices):} 
\[
T=\frac{1}{2}M_{ij}\dot{q}_i\dot{q}_j=\frac{1}{2}\dot{\mathbf{q}}^T\mathbf{M}\dot{\mathbf{q}},\quad
V=\frac{1}{2}K_{ij}q_iq_j=\frac{1}{2}\mathbf{q}^T\mathbf{K}\mathbf{q}
\]
\[
M_{ij}=\frac{\partial^2\mathcal{L}}{\partial\dot{q}_i\partial\dot{q}_j},\quad
K_{ij}=-\frac{\partial^2\mathcal{L}}{\partial q_i\partial q_j}
\]
\[
M_{ij}\ddot{q}_j+K_{ij}q_j=0.
\]

\textbf{Gen. Vel. \& Accel.:} 
\[
\dot{r}_a=\sum_i\frac{\partial r_a}{\partial q_i}\dot{q}_i,\quad
\ddot{r}_a=\sum_i\frac{\partial r_a}{\partial q_i}\ddot{q}_i+\sum_{i,j}\frac{\partial^2r_a}{\partial q_i\partial q_j}\dot{q}_i\dot{q}_j.
\]

\textbf{Gen. Force:} 
\[
Q_j=\sum_i\vec{F}_i\frac{\partial r_i}{\partial q_j}
=-\sum_i\vec{\nabla}_{r_i}V\frac{\partial \vec{r}_i}{\partial q_j}
=-\frac{\partial V}{\partial q_j}.
\]
\[
Q_j^{(nc)}=\sum_i\vec{F}_i^{(nc)}\cdot\frac{\partial \vec{r}_i}{\partial q_j}
=\sum_i\vec{F}_i^{(nc)}\cdot\frac{\partial \vec{v}_i}{\partial \dot{q}_j}.
\]

\textbf{Conservative \& Central Forces:} 
\[
\mathbf{F}=-\nabla V,\quad
\mathbf{F}=-\vec{\nabla}V(r)=-\frac{\partial V}{\partial r}\,\hat{r},
\]
\[
W=\int \mathbf{F}\cdot d\mathbf{r},\quad
\vec{J}=\int_{t_1}^{t_2}F\,dt.
\]

\textbf{Ang. Momentum \& Torque:}
\[
\mathbf{L}=\mathbf{r}\times\mathbf{p}=I\omega,\quad
\tau=\mathbf{r}\times\mathbf{F}=\frac{d\mathbf{L}}{dt}=\mathbf{I}\dot{\omega}.
\]

\textbf{Non-Inertial:} 
\[
\mathbf{F}_{\text{eff}}=\mathbf{F}-m\mathbf{a}_{\text{frame}}
-2m(\mathbf{\Omega}\times\mathbf{v})
-m\mathbf{\Omega}\times(\mathbf{\Omega}\times\mathbf{r}),
\]
\[
\mathbf{F}_{\text{centrifugal}}=-m\mathbf{\Omega}\times(\mathbf{\Omega}\times\mathbf{r}),
\quad
\mathbf{F}_{\text{Coriolis}}=-2m(\mathbf{\Omega}\times\mathbf{v}).
\]

%----------------------- II. Lagrange, Hamilton \& H-J -----------------------
\textbf{II. Lagrange, Hamilton \& H-J}\\[2pt]
\textbf{Lagrange Eq.:} 
\[
\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)
-\frac{\partial\mathcal{L}}{\partial q}=0,
\]
\[
\frac{\partial\mathcal{L}}{\partial q}
-\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)
+\frac{d^2}{dt^2}\left(\frac{\partial\mathcal{L}}{\partial\ddot{q}}\right)=0.
\]

\textbf{With Force:} 
\[
\frac{d}{dt}\left[\frac{\partial T}{\partial\dot{q}_j}\right]
-\frac{\partial T}{\partial q_j}
=-\frac{\partial V}{\partial q_j}=Q_j,
\]
\[
\frac{d}{dt}\left[\frac{\partial\mathcal{L}}{\partial\dot{q}_j}\right]
-\frac{\partial\mathcal{L}}{\partial q_j}
=Q_j^{(nc)}.
\]

\textbf{Rayleigh \& Dissipation:} 
\[
F=\frac{1}{2}\sum_{i,j}C_{ij}\dot{q}_i\dot{q}_j,\quad
F=\frac{1}{2}\sum_i C_i\dot{q}_i^2,
\]
\[
Q_i^{(\text{diss})}=-\frac{\partial F}{\partial\dot{q}_i}=-c\dot{x},
\]
\[
\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}_i}\right)
-\frac{\partial\mathcal{L}}{\partial q_i}
=-\frac{\partial F}{\partial\dot{q}_i},
\]
\[
P_{\text{diss}}=\sum_i Q_i^{(\text{diss})}\dot{q}_i=2F.
\]

\textbf{Cyclic Coord.:} 
\[
\frac{\partial\mathcal{L}}{\partial q_i}=0
\implies\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}_i}\right)=0
\implies p_i=\text{const.}
\]
\[
\mathcal{L}_{\text{eff}}=\mathcal{L}-p_i\dot{q}_i\,\,\texttt{(cyclic)}.
\]

\textbf{Canonical Mom.:} 
\[
p_i=\frac{\partial\mathcal{L}}{\partial\dot{q}_i},\quad
S[q(t)]=\int_{t_1}^{t_2}\mathcal{L}(\dot{q},q,t)\,dt.
\]

\textbf{Noether:} 
\[
\frac{d}{dt}\left(\sum_i\frac{\partial\mathcal{L}}{\partial\dot{q}_i}\frac{\partial f}{\partial\alpha}\right)=0,
\quad \frac{\partial\mathcal{L}}{\partial t}=0\Rightarrow E=\text{const.}
\]

\textbf{Hamiltonian:} 
\[
\mathcal{H}=\sum_i\dot{q}_ip_i-\mathcal{L},\quad
\dot{q}_i=\frac{\partial\mathcal{H}}{\partial p_i},\quad
\dot{p}_i=-\frac{\partial\mathcal{H}}{\partial q_i}.
\]
\[
\frac{d}{dt}f(p_i,q_i,t)
=\{ \mathcal{H},f \}+\frac{\partial f}{\partial t},
\quad \frac{d}{dt}\{ f,g \}
=\frac{\partial\{ f,g \}}{\partial t}+\{\{ f,g \},\mathcal{H}\}.
\]

\textbf{Hamiltonians:} 
\[
\mathcal{H}=\frac{p^2}{2m}+\frac{1}{2}kx^2\quad\text{(Harmonic Oscillator)},
\]
\[
\mathcal{H}=\frac{p^2}{2m}-\frac{GMm}{r}\quad\text{(Kepler Problem)}.
\]

\textbf{Poisson Brackets:} 
\[
\{f,g\}=\sum_i\left(\frac{\partial f}{\partial q_i}\frac{\partial g}{\partial p_i}
-\frac{\partial f}{\partial p_i}\frac{\partial g}{\partial q_i}\right),
\]
\[
q_i,\,q_j= \{p_i,p_j\}=0,\quad \{p_i,q_j\}=\delta_{ij},
\]
\[
\{A,B\}=-\{B,A\},\quad \{A,BC\}=B\{A,C\}+\{A,B\}C.
\]
\[
\{L_i,L_j\}=-\sum_k\epsilon_{ijk}L_k.
\]

\textbf{Hamilton--Jacobi:} 
\[
\frac{\partial S}{\partial t}+H\left(q_i,\frac{\partial S}{\partial q_i},t\right)=0.
\]

\det(K-\omega^2M)=0,\quad

\]
\[
t=\int_{x_0}^{x}\frac{dx}{\sqrt{\frac{2}{m}(E-V(x))}},\quad

%----------------------- IV. Orbits & Celestial -----------------------
\textbf{IV. Orbits \& Celestial}\\[2pt]
\textbf{Orbital Motion:} 
\[
\frac{d^2u}{d\theta^2}+u=-\frac{m}{L^2}\frac{dV}{du},
\]
\[
r=\frac{a(1-e^2)}{1+e\cos\theta},
\quad
r(\theta)=\frac{r_0}{1+\epsilon\cos(\theta-\theta_0)}
=\frac{a(1-\epsilon^2)}{1+\epsilon\cos(\theta-\theta_0)},
\]
\[
r_0=\frac{p_\theta^2}{\mu Gm_1m_2}=a(1-\epsilon^2),
\]
\[
r_{\min}=\frac{r_0}{1+\epsilon}=a(1-\varepsilon),\quad
r_{\max}=\frac{r_0}{1-\epsilon}=a(1+\varepsilon),
\]
\[
a=\frac{r_{\max}+r_{\min}}{2},\quad b=a\sqrt{1-\varepsilon^2},
\]
\[
\varepsilon=\sqrt{1+\frac{2E\,p_\theta^2}{G^2\mu M_1^2M_2^2}}
=\sqrt{1-\left(\frac{b}{a}\right)^2}.
\]
\[
E=\frac{1}{2}mv^2-\frac{GMm}{r},\quad
\dot{r}=\sqrt{\frac{2}{\mu}\left(E-V(r)-\frac{\ell^2}{2\mu r^2}\right)},
\]
\[
L=mr^2\dot{\theta}=\text{const.}
\]

\textbf{Kepler Two-Body:} 
\[
L=\frac{\mu}{2}\left(\dot{r}^2+r^2\dot{\phi}^2\right)-V(r),
\]
\[
\mu\ddot{r}=-\frac{dV}{dr}+\frac{p_\theta^2}{\mu r^3},\quad
\mu=\frac{m_1m_2}{m_1+m_2}.
\]
In COM frame:
\[
L=\frac{\mu\dot{r}^2}{2}-V(r),
\quad
L=\mathbf{r}\times\mu\dot{\mathbf{r}}=\text{const},
\]
\[
P_\phi=\frac{\partial L}{\partial\dot{\phi}}=\mu r^2\dot{\phi}=\text{const},
\]
\[
\mu\ddot{\mathbf{r}}=-\frac{GM\mu}{r^2}\hat{\mathbf{r}}
=-\frac{\partial V_{\text{eff}}}{\partial r},
\]
\[
V_{\text{eff}}(r)=V(r)+\frac{p_\theta^2}{2\mu r^2}
=-\frac{GM_1M_2}{r}+\frac{\ell^2}{2\mu r^2},
\]
\[
E=\frac{1}{2}\mu\dot{r}^2+V_{\text{eff}}(r),
\]
\[
\Delta\phi=2\int_{r_1}^{r_2}\frac{L\,dr}{r^2\sqrt{2\mu(E-V_{\text{eff}}(r))}},
\]
\[
\Delta\theta=2\int_{r_{\min}}^{\infty}\frac{l/r^2}{\sqrt{2\mu\left(E-V(r)-\frac{l^2}{2\mu r^2}\right)}}\,dr.
\]

\textbf{Laplace-Runge-Lenz:} 
\[
\mathbf{A}=\mathbf{p}\times\mathbf{L}-mk\,\hat{\mathbf{r}}.
\]

%----------------------- VI. Dimensional Analysis & Scaling -----------------------
\textbf{VI. Dim. Analysis \& Scaling}\\[2pt]
\textbf{Legendre Transform:} 
\[
\frac{df}{dx}=f'(x)=s(x)
\]
\[
g(s)=-s\cdot x(s)+f(x(s)).
\]
\textbf{Inverse Transform:} 
\[
V(x)=g(s(x))+f(x).
\]

\textbf{Homogeneity:} 
\[
f(ax)=a^kf(x),\quad E_{\text{Kin}}(a\vec{v})=\frac{1}{2}ma^2v^2,
\quad V(r)=g\,r^k.
\]
\[
\nabla f(x)\cdot x=kf(x),
\]
or
\[
x\cdot\frac{\partial f(x)}{\partial x}=kf(x).
\]

\textbf{Virial Theorem:} 
\[
2\langle T\rangle+\langle V\rangle=0,\quad 2<T>=k_{deg}\langle V\rangle.
\]

\textbf{Scaling:} 
\[
x\to\alpha x,\quad t\to\beta t,\quad v\to\frac{\alpha}{\beta}v,
\]
\[
\tilde{\mathcal{L}}=\Bigl(\frac{\alpha}{\beta}\Bigr)^2T-\alpha^kV,\quad
\Bigl(\frac{\alpha}{\beta}\Bigr)^2=\alpha^k\Rightarrow\beta=\alpha^{1-\frac{k}{2}},
\]
\[
\frac{v'}{v}=\Bigl(\frac{l'}{l}\Bigr)^{\frac{k}{2}},\quad
\frac{E'}{E}=\Bigl(\frac{l'}{l}\Bigr)^k,\quad
\Bigl(\frac{l'}{l}\Bigr)^{1-\frac{k}{2}}=\frac{t'}{t}.
\]

\textbf{Buckingham \(\Pi\) Thm:} 
\[
f(x_1,\ldots,x_n)=0\Rightarrow f(\Pi_1,\ldots,\Pi_{n-r})=0,
\]
\[
\tau\sim d^{1-\frac{k}{2}},\quad
v\sim d^{k/2},\quad
E\sim d^k,\quad
L\sim d^{1+\frac{k}{2}}.
\]
